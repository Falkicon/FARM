Interface SystemMetrics

Represents comprehensive system metrics data collected by the monitoring service. All numeric values are in their base units (bytes for memory/disk, milliseconds for time).

interface SystemMetrics {
    timestamp: string;
    system: {
        uptime: number;
        platform: string;
        hostname: string;
    };
    cpu: {
        usage: number;
        cores: number;
        temperature: number;
    };
    memory: {
        total: number;
        used: number;
        free: number;
        usedPercent: number;
    };
    network: {
        interfaces: {
            name: string;
            speed: number;
            type: string;
        }[];
        stats: {
            rx_bytes: number;
            tx_bytes: number;
            rx_sec: number;
            tx_sec: number;
        };
    };
    disk: {
        total: number;
        used: number;
        free: number;
        usedPercent: number;
    };
    services: {
        database: {
            connected: boolean;
            latency: null | number;
        };
        api: {
            status: "ok" | "error";
            responseTime: number;
        };
    };
}
